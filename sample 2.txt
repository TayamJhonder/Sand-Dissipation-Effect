/* -----------------------------------
   BACKGROUND PARTICLE ENGINE
----------------------------------- */
const bgCanvas = document.getElementById("bgCanvas");
const bgCtx = bgCanvas.getContext("2d");

bgCanvas.width = innerWidth;
bgCanvas.height = innerHeight;

const bgParticles = [];
const bgCount = 110;

// CURSOR POS
let mouse = { x: null, y: null, active: false };
let touchMode = false;
let touchTimeout = null;

class BgParticle {
    constructor() { this.reset(); }

    reset() {
        this.x = Math.random() * bgCanvas.width;
        this.y = Math.random() * bgCanvas.height;
        this.baseX = this.x;
        this.baseY = this.y;
        this.size = Math.random() * 3 + 1;
        this.speedX = Math.random() * 1 - 0.5;
        this.speedY = Math.random() * 1 - 0.5;
    }

    draw() {
        bgCtx.fillStyle = "rgba(0,200,255,0.9)";
        bgCtx.beginPath();
        bgCtx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        bgCtx.fill();
    }

    update() {
        // Normal motion
        this.x += this.speedX;
        this.y += this.speedY;

        if (this.x < 0 || this.x > bgCanvas.width) this.speedX *= -1;
        if (this.y < 0 || this.y > bgCanvas.height) this.speedY *= -1;

        if (mouse.active) {
            let dx = this.x - mouse.x;
            let dy = this.y - mouse.y;
            let dist = Math.sqrt(dx * dx + dy * dy);

            // Orbit around cursor
            if (dist < 120) {
                let angle = Math.atan2(dy, dx);
                this.x = mouse.x + Math.cos(angle) * 120;
                this.y = mouse.y + Math.sin(angle) * 120;
            }

            // Strong repulsion when very close
            if (dist < 50) {
                this.x += dx * 0.2;
                this.y += dy * 0.2;
            }
        }
    }
}

// INIT BACKGROUND PARTICLES
function initBgParticles() {
    bgParticles.length = 0;
    for (let i = 0; i < bgCount; i++) bgParticles.push(new BgParticle());
}
initBgParticles();

// BACKGROUND ANIMATION LOOP
function animateBg() {
    bgCtx.clearRect(0, 0, bgCanvas.width, bgCanvas.height);
    bgParticles.forEach(p => { p.update(); p.draw(); });
    requestAnimationFrame(animateBg);
}
animateBg();

/* -----------------------------------
   WELCOME PARTICLE ANIMATION
----------------------------------- */
const welcomeCanvas = document.getElementById("welcomeCanvas");
const welcomeCtx = welcomeCanvas.getContext("2d");

welcomeCanvas.width = innerWidth;
welcomeCanvas.height = innerHeight;

let welcomeParticles = [];
const welcomeParticleCount = 6000;

// CREATE OFFSCREEN CANVAS TO DRAW WELCOME
const welcomeTextCanvas = document.createElement("canvas");
const welcomeTextCtx = welcomeTextCanvas.getContext("2d");

// Dynamically calculate font size based on screen width - MUCH LARGER
function calculateWelcomeFontSize() {
    const screenWidth = window.innerWidth;
    if (screenWidth < 768) {
        return "bold 120px Arial";
    } else if (screenWidth < 1024) {
        return "bold 180px Arial";
    } else {
        return "bold 250px Arial";
    }
}

// Set canvas size based on screen
function setWelcomeCanvasSize() {
    const screenWidth = window.innerWidth;
    const screenHeight = window.innerHeight;
    
    welcomeTextCanvas.width = Math.min(screenWidth * 0.95, 1400);
    welcomeTextCanvas.height = Math.min(screenHeight * 0.7, 600);
}

setWelcomeCanvasSize();

// Draw text "WELCOME" with proper positioning
welcomeTextCtx.fillStyle = "white";
welcomeTextCtx.font = calculateWelcomeFontSize();
welcomeTextCtx.textAlign = "center";
welcomeTextCtx.textBaseline = "middle";
welcomeTextCtx.fillText("WELCOME", welcomeTextCanvas.width/2, welcomeTextCanvas.height/2);

let welcomeCoordinates = welcomeTextCtx.getImageData(0, 0, welcomeTextCanvas.width, welcomeTextCanvas.height);

// WELCOME PARTICLE CLASS
class WelcomeParticle {
    constructor(x, y) {
        // Start position random
        this.x = Math.random() * welcomeCanvas.width;
        this.y = Math.random() * welcomeCanvas.height;

        // Home position (text position)
        this.homeX = x;
        this.homeY = y;

        this.size = 2.5;
        this.baseSpeed = 0.1;
        this.density = Math.random() * 25 + 5;
        this.color = "#00c6ff"; // Blue color
    }

    draw() {
        welcomeCtx.fillStyle = this.color;
        welcomeCtx.fillRect(this.x, this.y, this.size, this.size);
    }

    update() {
        // Distance to mouse
        let dx = mouse.x - this.x;
        let dy = mouse.y - this.y;
        let dist = Math.sqrt(dx * dx + dy * dy);

        // WHEN MOUSE NEAR → REPEL (SAND BREAK)
        if (dist < 80) {
            let angle = Math.atan2(dy, dx);
            let force = -80 / dist;
            let fx = Math.cos(angle) * force * this.density;
            let fy = Math.sin(angle) * force * this.density;

            this.x += fx;
            this.y += fy;
        } else {
            // RETURN TO HOME POSITION (REFORM LETTER)
            let toHomeX = this.homeX - this.x;
            let toHomeY = this.homeY - this.y;

            this.x += toHomeX * 0.05;
            this.y += toHomeY * 0.05;
        }
    }
}

// GENERATE PARTICLES FROM WELCOME PIXELS
function initWelcomeParticles() {
    welcomeParticles = [];

    // Use a more efficient sampling to ensure complete text
    for (let y = 0; y < welcomeTextCanvas.height; y += 2) {
        for (let x = 0; x < welcomeTextCanvas.width; x += 2) {
            let index = (y * welcomeTextCanvas.width + x) * 4;
            if (welcomeCoordinates.data[index + 3] > 128) {
                let posX = x + welcomeCanvas.width / 2 - welcomeTextCanvas.width / 2;
                let posY = y + welcomeCanvas.height / 2 - welcomeTextCanvas.height / 2;
                
                // Only add if within visible canvas area
                if (posX > 0 && posX < welcomeCanvas.width && posY > 0 && posY < welcomeCanvas.height) {
                    welcomeParticles.push(new WelcomeParticle(posX, posY));
                }

                if (welcomeParticles.length >= welcomeParticleCount) return;
            }
        }
    }
}

initWelcomeParticles();

// WELCOME ANIMATION LOOP
function animateWelcome() {
    welcomeCtx.clearRect(0, 0, welcomeCanvas.width, welcomeCanvas.height);

    for (let p of welcomeParticles) {
        p.update();
        p.draw();
    }

    requestAnimationFrame(animateWelcome);
}
animateWelcome();

let contentRevealed = false;

function revealContent() {
    if (contentRevealed) return;
    
    contentRevealed = true;
    
    // Hide welcome canvas with fade effect
    welcomeCanvas.classList.add('canvas-hide');
    
    // Show all content with staggered animation
    const revealElements = document.querySelectorAll('.content-reveal');
    revealElements.forEach((element, index) => {
        setTimeout(() => {
            element.classList.add('show');
        }, index * 200); // Stagger the animations
    });
}

// CLICK TO REVEAL
window.addEventListener("click", (e) => {
    // Only trigger if clicking on the welcome section or body
    if (e.target.closest('section#home') || e.target === document.body) {
        revealContent();
    }
});

// TOUCH TO REVEAL
window.addEventListener("touchstart", (e) => {
    if (e.target.closest('section#home') || e.target === document.body) {
        revealContent();
    }
});

   EVENT LISTENERS

// MOUSE (PC) - RESTORED MOUSE EFFECT
window.addEventListener("mousemove", e => {
    if (touchMode) return;
    mouse.x = e.clientX;
    mouse.y = e.clientY;
    mouse.active = true;
});
window.addEventListener("mouseleave", () => {
    mouse.active = false;
    mouse.x = undefined;
    mouse.y = undefined;
});

// TOUCH (MOBILE)
window.addEventListener("touchstart", e => {
    touchMode = true;

    let t = e.touches[0];
    mouse.x = t.clientX;
    mouse.y = t.clientY;
    mouse.active = true;

    clearTimeout(touchTimeout);
    touchTimeout = setTimeout(() => {
        mouse.active = false;
        touchMode = false;
        mouse.x = undefined;
        mouse.y = undefined;
    }, 2000);
});

// RESIZE
window.addEventListener("resize", () => {
    bgCanvas.width = innerWidth;
    bgCanvas.height = innerHeight;
    welcomeCanvas.width = innerWidth;
    welcomeCanvas.height = innerHeight;
    
    // Recalculate font size and canvas dimensions
    setWelcomeCanvasSize();
    
    // Redraw the welcome text
    welcomeTextCtx.fillStyle = "white";
    welcomeTextCtx.font = calculateWelcomeFontSize();
    welcomeTextCtx.textAlign = "center";
    welcomeTextCtx.textBaseline = "middle";
    welcomeTextCtx.clearRect(0, 0, welcomeTextCanvas.width, welcomeTextCanvas.height);
    welcomeTextCtx.fillText("WELCOME", welcomeTextCanvas.width/2, welcomeTextCanvas.height/2);
    
    welcomeCoordinates = welcomeTextCtx.getImageData(0, 0, welcomeTextCanvas.width, welcomeTextCanvas.height);
    
    initBgParticles();
    initWelcomeParticles();
});
